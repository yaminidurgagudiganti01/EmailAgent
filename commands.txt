# Email Agent - Commands Cheatsheet
# =========================================================================
# All commands you need, in order. Run from the project root (emailagent/).
# =========================================================================


# -------------------------------------------------------------------------
# 0. ONE-TIME SYSTEM SETUP
# -------------------------------------------------------------------------

# Install uv (Python package manager) — pick one:
curl -LsSf https://astral.sh/uv/install.sh | sh           # macOS / Linux
# OR
powershell -c "irm https://astral.sh/uv/install.ps1 | iex" # Windows

# Get your OpenAI API key from: https://platform.openai.com/api-keys


# -------------------------------------------------------------------------
# 1. GMAIL OAUTH (one-time)
# -------------------------------------------------------------------------
# In Google Cloud Console (https://console.cloud.google.com/):
#   1. Create a project
#   2. APIs & Services -> Enable Gmail API
#   3. OAuth consent screen -> External -> add yourself as test user
#   4. Credentials -> Create OAuth client ID -> Desktop app -> download JSON
#   5. Save the downloaded file as: config/credentials.json
#
# First run of the agent will pop a browser for consent and cache the token
# at config/token.json automatically.


# -------------------------------------------------------------------------
# 2. PROJECT SETUP
# -------------------------------------------------------------------------

cd emailagent

# --- OPTION A: uv (recommended, faster) ---
uv venv
.venv\Scripts\activate              # Windows
# source .venv/bin/activate         # macOS/Linux
uv pip install -e ".[dev]"         # editable install + dev tools

# --- OPTION B: plain pip ---
python -m venv .venv
.venv\Scripts\activate              # Windows
# source .venv/bin/activate         # macOS/Linux
pip install --upgrade pip
pip install -e ".[dev]"

# Copy env template and fill in your values:
copy .env.example .env              # Windows
# cp .env.example .env              # macOS/Linux

# Minimum required in .env:
#   OPENAI_API_KEY=sk-...
#   OPENAI_MODEL=gpt-4o-mini        # or gpt-4o for better quality
#   USER_NAME=Your Name
#   USER_SIGNATURE=Best,\nYour Name


# -------------------------------------------------------------------------
# 3. RUN THE AGENT (CLI)
# -------------------------------------------------------------------------

# Standard run — triage unread mail, save drafts automatically to Gmail Drafts
python -m email_agent

# Approve mode — review each draft before saving (recommended for daily use)
python -m email_agent --approve

# Dry run — triage + draft but do NOT save anything to Gmail
python -m email_agent --dry-run

# Custom Gmail search query + limit
python -m email_agent --query "is:unread newer_than:1d" --max 25
python -m email_agent --query "from:boss@company.com" --max 5
python -m email_agent --query "label:important is:unread"
python -m email_agent --query "has:attachment is:unread newer_than:7d"

# Combine approve + custom query
python -m email_agent --approve --query "is:unread newer_than:1d" --max 20


# -------------------------------------------------------------------------
# 3b. RUN THE STREAMLIT UI
# -------------------------------------------------------------------------

streamlit run ui/app.py

# Custom port / headless (useful for remote servers)
streamlit run ui/app.py --server.port 8080 --server.headless true

# The UI lets you:
#   - configure query + max emails from the sidebar
#   - watch live progress of fetch / triage / draft
#   - edit each draft's subject + body inline before saving
#   - save drafts one-by-one or "Save all"
#   - regenerate any draft you don't like


# -------------------------------------------------------------------------
# 4. TESTS + LINTING
# -------------------------------------------------------------------------

# Run smoke tests (no OpenAI calls, no Gmail required)
pytest

# Run a specific test verbosely
pytest tests/test_smoke.py::test_graph_compiles -v

# Lint
ruff check src/ tests/

# Format
ruff format src/ tests/


# -------------------------------------------------------------------------
# 5. DEBUGGING
# -------------------------------------------------------------------------

# Check startup config (validates API key + credentials path)
python -c "from email_agent.config import settings; settings.validate(); print('Config OK')"

# Verify your OpenAI key works
python -c "from openai import OpenAI; c = OpenAI(); print(c.models.list().data[0].id)"

# Triage one fake email — no Gmail, no .env needed (uses OPENAI_API_KEY from env)
python -c "
from email_agent.schemas import Email
from email_agent.llm import triage_email
email = Email(
    id='1', thread_id='t1',
    from_addr='alice@example.com',
    subject='Coffee Thursday?',
    body='Hey! Want to grab coffee on Thursday around 3pm?',
)
print(triage_email(email))
"

# If Gmail auth is stuck, force re-auth:
del config\token.json               # Windows
# rm config/token.json              # macOS/Linux
python -m email_agent


# -------------------------------------------------------------------------
# 6. LOGS + IDEMPOTENCY STORE
# -------------------------------------------------------------------------

# View live log (tail the rotating log file)
Get-Content logs\email_agent.log -Wait -Tail 50    # Windows PowerShell
# tail -f logs/email_agent.log                      # macOS/Linux

# View full log
Get-Content logs\email_agent.log                   # Windows
# cat logs/email_agent.log                          # macOS/Linux

# Check how many emails are in the processed store
python -c "from email_agent.store import processed_count; print(processed_count(), 'emails processed')"

# Inspect the SQLite store directly
# (install sqlite3 CLI or use DB Browser for SQLite)
sqlite3 data\processed.db "SELECT * FROM processed_emails ORDER BY processed_at DESC LIMIT 20;"

# Reset the store (allows re-processing all emails — use with care)
del data\processed.db               # Windows
# rm data/processed.db              # macOS/Linux


# -------------------------------------------------------------------------
# 7. MAINTENANCE
# -------------------------------------------------------------------------

# Update dependencies
uv pip install -e ".[dev]" --upgrade

# Pin current versions to a lockfile
uv pip freeze > requirements.lock

# Switch model for higher quality drafts (edit .env):
#   OPENAI_MODEL=gpt-4o


# -------------------------------------------------------------------------
# 8. UNINSTALL / CLEAN
# -------------------------------------------------------------------------

deactivate                          # exit venv
rmdir /s .venv .pytest_cache .ruff_cache    # Windows
# rm -rf .venv .pytest_cache .ruff_cache    # macOS/Linux

del config\token.json               # revoke cached Gmail token
# To fully revoke Gmail access: https://myaccount.google.com/permissions
